import logging



[docs]def get_logger(key="", logname="-", level="none"):

    LEVELS = { 'all': logging.NOTSET,
               'debug':logging.DEBUG,
               'info':logging.INFO,
               'warning':logging.WARNING,
               'error':logging.ERROR,
               'critical':logging.CRITICAL,
               'none': 99,
               }

    if logname == "-":
        logname = None

    logging.basicConfig(
        format='%(asctime)s| %(levelname)-8s| %(name)s| %(message)s',
        datefmt='%Y/%m/%d %H:%M:%S',
        filename=logname,
        level=LEVELS.get(level, logging.NOTSET),
    )

    return logging.getLogger(key)
